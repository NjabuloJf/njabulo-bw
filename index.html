// AI Chatbot Class
class FrediAIChatbot {
    constructor(model = 'gpt-4') {
        this.model = model;
        this.context = [];
        this.temperature = 0.7;
        this.maxTokens = 150;
    }

    // Process user message
    async processMessage(message) {
        try {
            // Add message to context
            this.context.push({
                role: 'user',
                content: message,
                timestamp: new Date().toISOString()
            });

            // Generate AI response
            const response = await this.generateAIResponse();
            
            // Add response to context
            this.context.push({
                role: 'assistant',
                content: response,
                timestamp: new Date().toISOString()
            });

            return response;
        } catch (error) {
            console.error('Error processing message:', error);
            return 'I apologize, but I encountered an error.';
        }
    }

    async generateAIResponse() {
        // Simulated AI response generation
        const responses = [
            "Hello! How can I assist you today?",
            "I understand your question. Let me provide detailed information.",
            "Based on my analysis, I recommend the following approach.",
            "That's an interesting question! Here's what I think.",
            "I can help with that. Here's the information you need."
        ];
        
        return responses[Math.floor(Math.random() * responses.length)];
    }

    // Clear conversation context
    clearContext() {
        this.context = [];
        console.log('Chat context cleared');
    }
}

// Export the chatbot class
module.exports = FrediAIChatbot;

#!/usr/bin/env python3
# Universal Video Downloader

import yt_dlp
import os
from datetime import datetime
from pathlib import Path

class VideoDownloader:
    def __init__(self, output_dir='downloads'):
        """
        Initialize video downloader with custom options
        """
        self.output_dir = Path(output_dir)
        self.output_dir.mkdir(exist_ok=True)
        
        # YouTube DL options
        self.ydl_opts = {
            'format': 'best[ext=mp4]',
            'outtmpl': str(self.output_dir / '%(title)s.%(ext)s'),
            'quiet': False,
            'no_warnings': True,
            'ignoreerrors': True,
            'progress_hooks': [self.progress_hook],
        }
    
    def progress_hook(self, d):
        """Progress callback function"""
        if d['status'] == 'downloading':
            percent = d.get('_percent_str', '0%')
            speed = d.get('_speed_str', 'N/A')
            eta = d.get('_eta_str', 'N/A')
            print(f"\rDownloading: {percent} | Speed: {speed} | ETA: {eta}", end='')
        elif d['status'] == 'finished':
            print(f"\nDownload completed: {d['filename']}")
    
    def download_video(self, url):
        """
        Download video from URL
        """
        try:
            print(f"Starting download from: {url}")
            
            with yt_dlp.YoutubeDL(self.ydl_opts) as ydl:
                info = ydl.extract_info(url, download=True)
                
                if info:
                    filename = ydl.prepare_filename(info)
                    file_size = os.path.getsize(filename) if os.path.exists(filename) else 0
                    
                    result = {
                        'success': True,
                        'filename': filename,
                        'title': info.get('title', 'Unknown'),
                        'duration': info.get('duration', 0),
                        'file_size': file_size,
                        'timestamp': datetime.now().isoformat()
                    }
                    return result
                    
        except Exception as e:
            print(f"Error downloading video: {e}")
            return {
                'success': False,
                'error': str(e)
            }
    
    def get_video_info(self, url):
        """
        Get video information without downloading
        """
        try:
            with yt_dlp.YoutubeDL({'quiet': True}) as ydl:
                info = ydl.extract_info(url, download=False)
                
                video_info = {
                    'title': info.get('title'),
                    'duration': info.get('duration'),
                    'uploader': info.get('uploader'),
                    'view_count': info.get('view_count'),
                    'upload_date': info.get('upload_date'),
                    'thumbnail': info.get('thumbnail'),
                    'formats': []
                }
                
                # Get available formats
                for fmt in info.get('formats', []):
                    video_info['formats'].append({
                        'format_id': fmt.get('format_id'),
                        'ext': fmt.get('ext'),
                        'resolution': fmt.get('resolution'),
                        'filesize': fmt.get('filesize')
                    })
                
                return video_info
                
        except Exception as e:
            print(f"Error getting video info: {e}")
            return None

# Usage example
if __name__ == "__main__":
    downloader = VideoDownloader()
    
    # Example URL
    url = "https://www.youtube.com/watch?v=dQw4w9WgXcQ"
    
    # Get video info
    info = downloader.get_video_info(url)
    if info:
        print(f"Video Title: {info['title']}")
        print(f"Duration: {info['duration']} seconds")
        print(f"Available formats: {len(info['formats'])}")
    
    # Download video
    result = downloader.download_video(url)
    if result['success']:
        print(f"Download successful: {result['filename']}")

import React, { useState, useEffect, useCallback } from 'react';
import axios from 'axios';
import debounce from 'lodash/debounce';
import './SearchComponent.css';

const SearchComponent = ({ apiEndpoint, placeholder = "Search..." }) => {
    const [query, setQuery] = useState('');
    const [results, setResults] = useState([]);
    const [loading, setLoading] = useState(false);
    const [error, setError] = useState(null);
    const [searchHistory, setSearchHistory] = useState([]);

    // Initialize from localStorage
    useEffect(() => {
        const savedHistory = localStorage.getItem('searchHistory');
        if (savedHistory) {
            setSearchHistory(JSON.parse(savedHistory));
        }
    }, []);

    // Debounced search function
    const performSearch = useCallback(
        debounce(async (searchQuery) => {
            if (!searchQuery.trim()) {
                setResults([]);
                return;
            }

            setLoading(true);
            setError(null);

            try {
                const response = await axios.get(apiEndpoint, {
                    params: { q: searchQuery, limit: 10 }
                });

                setResults(response.data.results || response.data);
                
                // Update search history
                const newHistory = [
                    { query: searchQuery, timestamp: new Date().toISOString() },
                    ...searchHistory.filter(item => item.query !== searchQuery)
                ].slice(0, 10);
                
                setSearchHistory(newHistory);
                localStorage.setItem('searchHistory', JSON.stringify(newHistory));
                
            } catch (err) {
                setError('Failed to fetch search results. Please try again.');
                console.error('Search error:', err);
            } finally {
                setLoading(false);
            }
        }, 500),
        [apiEndpoint, searchHistory]
    );

    // Trigger search when query changes
    useEffect(() => {
        performSearch(query);
    }, [query, performSearch]);

    // Clear search results
    const clearSearch = () => {
        setQuery('');
        setResults([]);
        setError(null);
    };

    // Handle search history click
    const handleHistoryClick = (historyQuery) => {
        setQuery(historyQuery);
    };

    // Clear search history
    const clearHistory = () => {
        setSearchHistory([]);
        localStorage.removeItem('searchHistory');
